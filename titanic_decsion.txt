import pandas as pd
from sklearn.tree import DecisionTreeClassifier, plot_tree
import matplotlib.pyplot as plt

data = pd.read_csv("titanic.csv")

X = data[['Pclass','Sex','Age']]
y = data['Survived']

model = DecisionTreeClassifier(random_state=0)
model.fit(X, y)

# prediction example
prediction = model.predict([[3,0,25]])

print("Passenger Prediction (0=Not Survived, 1=Survived):", prediction[0])

# visualize decision tree
plt.figure(figsize=(8,5))
plot_tree(model, feature_names=['Pclass','Sex','Age'], class_names=['No','Yes'], filled=True)
plt.title("Titanic Decision Tree")
plt.show()
