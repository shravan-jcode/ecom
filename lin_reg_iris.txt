import pandas as pd
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.linear_model import LinearRegression

# Load built-in Iris dataset
iris = load_iris()

# Convert dataset to DataFrame
data = pd.DataFrame(iris.data, columns=iris.feature_names)

# Independent variable (petal length)
X = data[["petal length (cm)"]]

# Dependent variable (petal width)
y = data["petal width (cm)"]

# Create Linear Regression model
model = LinearRegression()

# Train the model
model.fit(X, y)

# Predict petal width for a new petal length
value = float(input("Enter Petal Length: "))
prediction = model.predict([[value]])

print("Predicted Petal Width:", round(prediction[0],2))

# Plot graph
plt.scatter(X, y)
plt.plot(X, model.predict(X))
plt.xlabel("Petal Length")
plt.ylabel("Petal Width")
plt.title("Linear Regression on Iris Dataset")
plt.show()
