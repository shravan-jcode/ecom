from scipy import stats

# Exam scores of students
class_A = [85, 90, 88, 82, 87]
class_B = [76, 78, 80, 81, 75]
class_C = [92, 88, 94, 89, 90]

# Perform One Way ANOVA Test
f_statistic, p_value = stats.f_oneway(class_A, class_B, class_C)

print("F-Statistic:", f_statistic)
print("P-Value:", p_value)

# Decision making
alpha = 0.05

if p_value < alpha:
    print("Reject Null Hypothesis")
    print("There is a significant difference in mean exam scores among the classes.")
else:
    print("Fail to Reject Null Hypothesis")
    print("There is no significant difference in mean exam scores among the classes.")
